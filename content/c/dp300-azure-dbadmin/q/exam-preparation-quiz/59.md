answers:
    - id: "answer1"
      title: "A filegroup"
      correct: false
      explain: "Filegroups are not required for partitioned tables in Azure SQL Database, which uses a single filegroup."
    - id: "answer2"
      title: "A partition function"
      correct: true
      explain: "The partition function must be created first as it defines how rows will be distributed across partitions based on column values."
    - id: "answer3"
      title: "An aligned index"
      correct: false
      explain: "Aligned indexes are created after the table exists, not before creating the partitioned table."
    - id: "answer4"
      title: "A partition scheme"
      correct: true
      explain: "The partition scheme is created second and maps the partitions defined by the partition function to filegroups (or PRIMARY in Azure SQL Database)."
    - id: "answer5"
      title: "A table"
      correct: true
      explain: "The table is created last, using the partition scheme to distribute data across partitions."
