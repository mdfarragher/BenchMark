title: "Allow users to create database objects without modifying existing ones"

answers:
    - id: "answer1"
      title: "For each customer, grant the customer ddl_admin to the existing schema"
      correct: false
      explain: "Granting ddl_admin on the existing schema would allow users to modify existing objects, which violates the requirement."
    - id: "answer2"
      title: "For each customer, create an additional schema and grant the customer ddl_admin to the new schema"
      correct: true
      explain: "Creating a separate schema for each customer and granting ddl_admin only to that schema allows users to create objects without being able to modify existing objects in other schemas."
    - id: "answer3"
      title: "For each customer, create an additional schema and grant the customer db_writer to the new schema"
      correct: false
      explain: "The db_writer role allows data modification but not object creation. You need ddl_admin for creating database objects."
    - id: "answer4"
      title: "For each customer, grant the customer db_writer to the existing schema"
      correct: false
      explain: "The db_writer role allows data modification but not object creation, and doesn't prevent modification of existing objects."
