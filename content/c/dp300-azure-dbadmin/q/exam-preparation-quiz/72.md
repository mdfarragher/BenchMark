title: "Track Column Access and Store Audit Logs for 365 Days"

layout: "multiple"
answers:
    - id: "answer1"
      title: "Turn on auditing and write audit logs to an Azure Storage account"
      correct: true
      explain: "Azure Storage accounts can retain audit logs for 365 days and provide cost-effective long-term storage for audit data."
    - id: "answer2"
      title: "Add extended properties to the column"
      correct: false
      explain: "Extended properties are metadata tags but don't provide auditing or tracking of column access."
    - id: "answer3"
      title: "Turn on auditing and write audit logs to an Event Hub"
      correct: false
      explain: "Event Hub is designed for real-time streaming and is not suitable for long-term audit log storage for 365 days."
    - id: "answer4"
      title: "Apply sensitivity labels named Highly Confidential to the column"
      correct: true
      explain: "Sensitivity labels enable auditing of access to sensitive columns, allowing you to track when values are returned in queries."
    - id: "answer5"
      title: "Turn on Azure Defender for SQL"
      correct: false
      explain: "While Azure Defender provides threat protection, it doesn't specifically track column-level access or provide the required audit logging."
